class Histogram{
  constructor (context, unit, k, x, y, width, height) {
    this.context = context
    this.unit = unit
    this.k = k
    this.x = x 
    this.y = y
    this.height = height
    this.width = width
    this.box = new Rectangle(context, x, y, width, height)
    this.histo = []
    this.calcIntervals()
  }

  calcIntervals() {
    var bin = -this.unit/2
    for (var i = 0; i <= this.k; i ++) {
      this.box.ctx.fillText(bin, this.x + this.width, this.y + this.height - i/this.k * this.height);
      this.histo[i] = [bin, 0]
      bin = bin + this.unit/this.k
    }


    var scaledY = 0
    for (var i = 0; i < 1; i = i + 1/this.k) {
      scaledY = i * this.height
      this.box.drawLine(0, scaledY, this.width, scaledY, "Lightgrey")
    }
  }

  addToBin(v) {
    for (var i = 0; i < this.k; i++) {
      if (v > this.histo[i][0] && v <= this.histo[i+1][0]){
        this.histo[i][1] ++;
        this.box.drawRect(0, this.height - (i+1)*(1/this.k * this.height), this.histo[i][1]*5.5, 1/this.k* this.height, "Red")
      }
    }
  }}

## Simulation of cyberattacks to some servers
In this homework we will simulate M servers that are getting attacked N times.
We will draw a cartesia graph where on the *X-axis* we indicate the attacks and on the *Y-axis* we simulate the accumulation of the *security score*.
An attack succeed with probability p = 1/2 (and thus fail with p=1/2) in each simulation.

### Simulation 1: (1,-1) security score
In this simulation we decrement the cumulated security score of a server by -1 if an attack succeed, and increment it by 1 when an attack fails. So the higher the security score, the better the system held against the N attacks.

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms.VisualStyles;

namespace attacks_simulation
{
    internal class Simulator
    {
        private int nSystems;
        private int nAttacks;
        private int successPoints;
        private int failurePoints;
        private double p;
        private String frequencyType = "global";
        private int midDay;
        private float x;
        private float y;
        private Graphics graphics;
        private Box box;
        private float boxWidth = 800;
        private float boxHeight = 600;
        private float rowDistance;
        private float colDistance;
        private float stepX;
        private float stepY;
        Histogram h1, h2;
        private int k;



        public Simulator(int nSystems, int nAttacks, int successPoints, int failurePoints, double p, String frequencyType, int midDay, float x, float y, Graphics graphics, int k) 
        { 
            this.nSystems = nSystems;
            this.nAttacks = nAttacks;
            this.successPoints = successPoints;
            this.failurePoints = failurePoints;
            this.p = p;
            this.frequencyType = frequencyType;
            this.midDay = midDay;
            this.x = x;
            this.y = y;
            this.graphics = graphics;
            this.rowDistance = this.boxHeight / this.nAttacks;
            this.colDistance = this.boxWidth / this.nAttacks;
            this.stepX = this.colDistance;
            this.stepY = this.rowDistance / 2;
            this.box = new Box(graphics, x, y, this.boxWidth, this.boxHeight);
            this.k = k;
            
        }

        public void drawChart() 
        {
            float rowDistance = this.boxHeight / this.nAttacks;
            float colDistance = this.boxWidth / this.nAttacks;
            for (float i = rowDistance/2; i < this.boxHeight; i = i + rowDistance/2) {
                this.box.drawLine(0, i, this.boxWidth, i, Color.LightGray);
            }
            for (float i = colDistance; i < this.boxWidth; i = i + colDistance)
            {
                this.box.drawLine(i, 0, i, this.boxHeight, Color.LightGray);
            }
            
            this.h1 = new Histogram(this.graphics, this.k, this.x + this.midDay * this.colDistance, this.y, 30, this.boxHeight);
            this.h1.drawChart();
            this.h2 = new Histogram(this.graphics, this.k, this.x+this.boxWidth, this.y, 30, this.boxHeight);
            this.h2.drawChart();


        }

        public void simulate()
        {
            Random r = new Random();
            Random rnd = new Random();
            System.Timers.Timer t = new System.Timers.Timer();
            t.Interval = 2000;
            for (int i = 0; i < this.nSystems; i++) 
            {
                float x1 = 0;
                float y1 = this.boxHeight / 2;
                float x2 = 0;
                float y2 = this.boxHeight / 2;
                int esito = 0;
                double count = 0;
                double s = 0;
                Color randomColor = Color.FromArgb(rnd.Next(256), rnd.Next(256), rnd.Next(256));

                for (int j = 0; j < this.nAttacks; j++) 
                {   
                    if (r.NextDouble() >= this.p)
                    {
                        esito = esito + this.successPoints;
                    }
                    else
                    {
                        esito = esito + this.failurePoints;
                    }

                    count += 1;
                    if (this.frequencyType.Equals("rel"))
                    {
                        s = (esito / count) * 10;
                    }
                    else if (this.frequencyType.Equals("reln"))
                    {
                        s = (esito / Math.Sqrt(count)) * 4;
                    }
                    else 
                    {
                        s = esito;
                    }

                    if (j == this.midDay) 
                    {
                        this.h1.addToBin((int) s);
                    }


                    x2 = x1 + this.stepX;
                    y2 = (float) ((this.boxHeight/2) - (this.stepY * s));
                    this.box.drawLine(x1, y1, x2, y2, randomColor);
                    x1 = x2;
                    y1 = y2;
                }
                this.h2.addToBin((int) s);
            }
        }
    }
}

        {
            this.graphics = graphics;
            this.x = x;
            this.y = y;
            this.width = width;
            this.height = heigth;
            Pen pen = new Pen(Color.Black, 2);
            this.graphics.DrawRectangle(pen, this.x, this.y, this.width, this.height);
        }

        public void drawLine(float x1, float y1, float x2, float y2, Color color) 
        {
            Pen pen = new Pen(color, 2);
            float relative_x1 = this.x + x1;
            float relative_y1 = this.y + y1;

            float relative_x2 = this.x + x2;
            float relative_y2 = this.y + y2;

            this.graphics.DrawLine(pen, relative_x1, relative_y1, relative_x2, relative_y2);
        }

        public void drawRect(float x, float y, float width, float height, Color color) 
        {
            SolidBrush brush = new SolidBrush(color);
            this.graphics.FillRectangle(brush, this.x + x, this.y + y, width, height);
        }

using System;
using System.Collections.Generic;
using System.Linq;
using System.Runtime.CompilerServices;
using System.Text;
using System.Threading.Tasks;
using System.Xml.Linq;

namespace attacks_simulation
{
    internal class Histogram
    {
        private int k;
        private float x;
        private float y;
        private float width;
        private float height;
        private Box box;
        private float distance;
        private float[,] histoValues;

        public Histogram(Graphics graphic, int k, float x, float y, float width, float height)
        {
            this.k = k;
            this.x = x;
            this.y = y;
            this.width = width;
            this.height = height;
            this.box = new Box(graphic, x, y, this.width, this.height);
            this.distance = this.height/this.k;
            this.histoValues = new float[k, 100];

            int c_bin = -this.k / 2;
            for (int i = 0; i < this.k; i++) 
            {
                this.box.graphics.DrawString((-c_bin*2).ToString(), new Font("Tahoma", 8), Brushes.Black, this.x + this.width, i*this.distance);
                histoValues[i,0] = (-c_bin*2);
                c_bin++;
            }

        }

        public void drawChart() 
        { 
            for (int i = 0; i < this.height; i += (int) this.distance) 
            {
                this.box.drawLine(0, i, this.width, i, Color.Black);
            }
        }

        public void addToBin(int value)
        {
            for (int i = 0; i < this.k-1; i++)
            {
                if (value > this.histoValues[i + 1, 0] && value <= this.histoValues[i, 0])
                {
                    this.histoValues[i, 1] += 1;
                    this.box.drawRect(this.width,  i * this.distance, this.histoValues[i, 1]*4, this.distance, Color.Red);
                }
            }

        }
    }
}

## Excercise 2
A *probability space* is a mathematical object that models the execution of volatile experiments. It is a triple  ( Ω , F , P ) where:
- Ω is the *sample space*, or the set of all possible outcomes
- F is an *event space*, or a set of events, where an event is a set of outcomes
- P is a *probability function* wich assigns to each event space a probability, a number between 0 and 1.
Some examples of probability spaces are:
- The throwing of a dice, where we want to study the probability of the landing on a even number:
    - Ω = \{1,2,3,4,5,6\}
    - F = \{2,4,6\}
    - P(\{2,4,6\}) = 3/6
- The throwing of a coin where:
    - Ω = \{H, T\}
    - F = \{\{H\}, \{T\}\}
    - P(H) = 1/2 ; P(T) = 1/2

In the case of excercise 2 the triple is:
- Ω = \{SUCCESS, FAILURE\}
- F = \{\{SUCCESS\}, \{FAILURE\}\}
- P(SUCCESS) = p ; P(FAILURE) = 1-p
