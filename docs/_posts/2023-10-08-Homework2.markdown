<script>
    class Dictionary {
      constructor() {
        this.data = {};
      }

      // Add a key-value pair to the dictionary
      add(key, value) {
        this.data[key] = value;
      }

      // Retrieve a value by key
      get(key) {
        return this.data[key];
      }

      // Check if a key exists in the dictionary
      has(key) {
        return key in this.data;
      }

      // Remove a key-value pair from the dictionary
      remove(key) {
        if (this.has(key)) {
          delete this.data[key];
        }
      }

      // Get all keys in the dictionary
      keys() {
        return Object.keys(this.data);
      }

      // Get all values in the dictionary
      values() {
        return Object.values(this.data);
      }

      // Get the number of key-value pairs in the dictionary
      size() {
        return Object.keys(this.data).length;
      }

      // Clear the dictionary
      clear() {
        this.data = {};
      }
    }
</script>

<script>
    class SortedSet {
      constructor(compareFunction) {
        this.set = new Set();
        this.compare = compareFunction || ((a, b) => a - b);
      }

      // Add an element to the sorted set
      add(element) {
        if (!this.has(element)) {
          this.set.add(element);
          this.sort();
        }
      }

      // Delete an element from the sorted set
      delete(element) {
        if (this.has(element)) {
          this.set.delete(element);
        }
      }

      // Check if an element exists in the sorted set
      has(element) {
        return this.set.has(element);
      }

      // Get the size (number of elements) of the sorted set
      size() {
        return this.set.size;
      }

      // Get the elements of the sorted set as an array
      toArray() {
        return Array.from(this.set);
      }

      // Sort the elements in the set
      sort() {
        const array = this.toArray();
        array.sort(this.compare);
        this.set = new Set(array);
      }
    }
</script>

<script>
    class Queue {
      constructor() {
        this.items = [];
      }

      // Add an element to the end of the queue
      enqueue(element) {
        this.items.push(element);
      }

      // Remove and return the element at the front of the queue
      dequeue() {
        if (this.isEmpty()) {
          return null;
        }
        return this.items.shift();
      }

      // Return the element at the front of the queue without removing it
      front() {
        if (this.isEmpty()) {
          return null;
        }
        return this.items[0];
      }

      // Check if the queue is empty
      isEmpty() {
        return this.items.length === 0;
      }

      // Get the number of elements in the queue
      size() {
        return this.items.length;
      }

      // Clear the queue

        this.items = [];
      }

    class Stack {
      constructor() {
        this.items = [];
      }

      // Push an element onto the stack
      push(element) {
        this.items.push(element);
      }

      // Pop and return the element from the top of the stack

        if (this.isEmpty()) {

#### Linkedlists
{% highlight javascript %}
<script>
    class Node {
      constructor(data) {
        this.data = data;
        this.next = null;
      }
    }

    class LinkedList {
      constructor() {
        this.head = null;
        this.size = 0;
      }

      // Add an element to the end of the linked list
      add(data) {
        const newNode = new Node(data);
        if (!this.head) {
          this.head = newNode;
        } else {
          let current = this.head;
          while (current.next) {
            current = current.next;
          }
          current.next = newNode;
        }
        this.size++;
      }

      // Remove an element by value from the linked list
      remove(data) {
        if (!this.head) {
          return;
        }
        if (this.head.data === data) {
          this.head = this.head.next;
          this.size--;
          return;
        }

        let current = this.head;
        let previous = null;

        while (current && current.data !== data) {
          previous = current;
          current = current.next;
        }

        if (current) {
          previous.next = current.next;
          this.size--;
        }
      }

      // Get the size (number of elements) of the linked list
      getSize() {
        return this.size;
      }
    }
</script>
