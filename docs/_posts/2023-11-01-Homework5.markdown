<script>
class Histogram{
  constructor (context, unit, k, x, y, width, height) {
    this.context = context
    this.unit = unit
    this.k = k
    this.x = x 
    this.y = y
    this.height = height
    this.width = width
    this.box = new Rectangle(context, x, y, width, height)
    this.histo = []
    this.calcIntervals()
  }

  calcIntervals() {
    var bin = -this.unit/2
    for (var i = 0; i <= this.k; i ++) {
      this.box.ctx.fillText(bin, this.x + this.width, this.y + this.height - i/this.k * this.height);
      this.histo[i] = [bin, 0]
      bin = bin + this.unit/this.k
    }


    var scaledY = 0
    for (var i = 0; i < 1; i = i + 1/this.k) {
      scaledY = i * this.height
      this.box.drawLine(0, scaledY, this.width, scaledY, "Lightgrey")
    }
  }

  addToBin(v) {
    for (var i = 0; i < this.k; i++) {
      if (v > this.histo[i][0] && v <= this.histo[i+1][0]){
        this.histo[i][1] ++;
        this.box.drawRect(0, this.height - (i+1)*(1/this.k * this.height), this.histo[i][1]*5.5, 1/this.k* this.height, "Red")
      }
    }
  }}
</script>
<script>
const width = 700 //rectangle width
const height = 400 //rectangle height
class Simulator {
  constructor(context, x, y, N, M, success, failure, frequency, probability) {
    this.context = context
    this.x = x //x position of rectangle
    this.y = y //y position of rectangle
    this.N = N //number of intervals over 1
    this.M = M //number tries
    this.success = success //increment for success
    this.failure = failure //increment for failure
    this.frequency = frequency // type of frequency calculated
    this.probability = probability // probability of attack
    this.box = new Rectangle(context, x, y, width, height)
    this.histogramMiddle = new Histogram(context, 2*N ,50, this.x + 350, this.y, 100, height)
    this.histogramLast = new Histogram(context, 2*N, 50, this.x + width, this.y, 100, height)
    this.calcIntervals()
  }

  calcIntervals() {
    var scaledX = 0
    var scaledY = 0
    for (var i = 1/this.N; i < 1; i = i + 1/this.N){
      scaledX = i * width
      this.box.drawLine(scaledX, 0, scaledX, height, "Lightgrey")
    }
    for (var i = 1/this.N; i < 1; i = i + 1/this.N){
      scaledY = i * height
      this.box.drawLine(0, scaledY, width, scaledY, "Lightgrey")
    }
  }

  simulate(){
    var color = ""
    var step = 0
    var rand = 0
    var x2 = 0
    var y2 = height/2
    var outcome = 0
    var div = 1
    for (var trial = 0; trial < this.M; trial++){
      var x1 = 0
      var y1 = height/2
      var y2 = height/2
      var total = 0;
      color = "#"+((1<<24)*Math.random()|0).toString(16);
      for (var time = 1; time <= this.N; time++) {

        if (Math.random() < this.probability) {
          outcome = this.success
        } else {
          outcome = this.failure
        }

        if (this.frequency == "relative") {
          div = time
        }
        if (this.frequency == "nRelative") {
          div = Math.sqrt(time)
        }

        total += outcome/div


        if (time == 25) {
          this.histogramMiddle.addToBin(total)
        }

        x2 = x1 + 1/this.N * width
        step =  height * 1/this.N/2 * -outcome/div
        y2 = y1 + step
        this.box.drawLine(x1, y1, x2, y2, color)
        x1 = x2
        y1 = y2
      }
      this.histogramLast.addToBin(total)
    }
  }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace hw5
{
    internal class Histogram
    {
        private int k;
        private float x;
        private float y;
        private float width;
        private float height;
        private Box box;
        private float distance;
        private float[,] histoValues;

        public Histogram(Graphics graphic, int k, float x, float y, float width, float height)
        {
            this.k = k;
            this.x = x;
            this.y = y;
            this.width = width;
            this.height = height;
            this.box = new Box(graphic, x, y, this.width, this.height);
            this.distance = this.height / this.k;
            this.histoValues = new float[k, 100];

            int c_bin = 0;
            for (int i = 0; i < this.k; i++)
            {
                this.box.graphics.DrawString((c_bin).ToString(), new Font("Tahoma", 8), Brushes.Black, this.x + this.width, this.height - i * this.distance);
                histoValues[i, 0] = (c_bin);
                c_bin++;
            }

        }

        public void drawChart()
        {
            for (int i = 0; i < this.height; i += (int)this.distance)
            {
                this.box.drawLine(0, i, this.width, i, Color.Black);
            }
        }

        public void addToBin(int value)
        {
            for (int i = 0; i < this.k - 1; i++)
            {
                if (value < this.histoValues[i + 1, 0] && value >= this.histoValues[i, 0])
                {
                    this.histoValues[i, 1] += 1;
                    this.box.drawRect(this.width, (this.k-i) * this.distance, this.histoValues[i, 1] * 4, this.distance, Color.Red);
                }
            }

        }
    }
}

using System;

    internal class Box
    {
        float x, y, width, height;
        public Graphics graphics;
        public Box(Graphics graphics, float x, float y, float width, float heigth)
        {
            this.graphics = graphics;
            this.x = x;
            this.y = y;
            this.width = width;
            this.height = heigth;
            Pen pen = new Pen(Color.Black, 2);
            this.graphics.DrawRectangle(pen, this.x, this.y, this.width, this.height);
        }

        public void drawLine(float x1, float y1, float x2, float y2, Color color)
        {
            Pen pen = new Pen(color, 2);
            float relative_x1 = this.x + x1;
            float relative_y1 = this.y + y1;

            float relative_x2 = this.x + x2;
            float relative_y2 = this.y + y2;

            this.graphics.DrawLine(pen, relative_x1, relative_y1, relative_x2, relative_y2);
        }

        public void drawRect(float x, float y, float width, float height, Color color)
        {
            SolidBrush brush = new SolidBrush(color);
            this.graphics.FillRectangle(brush, this.x + x, this.y + y, width, height);
        }
